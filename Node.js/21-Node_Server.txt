HTTP Node:
    We Use Nodes HTTP module to create a server
    
const http = require("node:http");

//Always speccify content type
const server = http.createServer((req,res) => {
    res.writeHead(200,{"Content-Type":"text/plain"}); //Successfull response
    res.end("Hello World!"); //Text Response
});

server.listen(3000,() =>{
    console.log(`Server running on port 3000`);
}); //Port no
